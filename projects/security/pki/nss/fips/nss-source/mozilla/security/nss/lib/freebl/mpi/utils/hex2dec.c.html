   49 int main(int argc, char *argv[])
   50 {
   51   mp_int  a;
   52   char   *buf;
   53   int     len;
   54 
   55   if(argc < 2) {
   56     fprintf(stderr, "Usage: %s <a>\n", argv[0]);
   57     return 1;
   58   }
   59 
   60   mp_init(&a); mp_read_radix(&a, argv[1], 16);
   61   len = mp_radix_size(&a, 10);
   62   buf = malloc(len);
   63   mp_toradix(&a, buf, 10);
   64 
   65   printf("%s\n", buf);
   66 
   67   free(buf);
   68   mp_clear(&a);
