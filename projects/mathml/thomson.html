\begin{eqnarray*}
\frac{d^2P}{d\Omega_s d\omega_s}
& = &
r^2_e 
\int_V <S_i> d^3 r 
\int | \hat{e} . 
\mathop{\Pi}\limits^\leftrightarrow .
\hat{e} |^2 \kappa^2 f \delta(k . v - w) d^3 v\\
& = & 
r^2_e 
\int_V <S_i> d^3 r 
\int \left| 1 - 
\frac{1 - \hat s . \hat i}{(1 - \beta_i)(1 - \beta_s)}
\beta_e^2\right|^2
\left|\frac{1 - \beta_i}{1 - \beta_s}\right|^2 \\
&  &
\times (1 - \beta^2) f \delta( k . b - w) d^3 v
\end{eqnarray*}
