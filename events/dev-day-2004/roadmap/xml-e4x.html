XML with E4X

function createTable() { var doc = XML(document); // create an E4X wrapper for the doc var tbody = doc..body.TABLE.TBODY; for (var j = 0; j < 2; j++) { tbody.TR[j] = ""; // append an empty table row for (var i = 0; i < 2; i++) // append a cell with some content tbody.TR[j].TD[i] = "cell is row " + j + ", column " + i; } doc..body.TABLE.@border = 2; // set the table's border attr }
