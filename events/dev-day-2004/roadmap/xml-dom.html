function createTable () { var body = document.getElementsByTagName("body").item(0); var table = document.createElement("TABLE"); var tbody = document.createElement("TBODY"); for (var j = 0; j < 2; j++) { var cur_row = document.createElement("TR"); for (var i = 0; i < 2; i++) { var cur_cell = document.createElement("TD"); var cur_text = document.createTextNode("cell is row "+j+", column "+i); cur_cell.appendChild(cur_text); cur_row.appendChild(cur_cell); } tbody.appendChild(cur_row); } table.appendChild(tbody); body.appendChild(table); table.setAttribute("border","2"); }
