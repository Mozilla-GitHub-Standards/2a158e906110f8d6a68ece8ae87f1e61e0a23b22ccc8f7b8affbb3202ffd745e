Getting the Element

function getElt () 
{ if (is.nav4up) {
   var currentLayer = document.layers[getElt.arguments[0]];
   for (var i=1; i<getElt.arguments.length 
        && currentLayer; i++)
   {   currentLayer = currentLayer.document.layers
                        [getElt.arguments[i]];      }
   return currentLayer;
  } 
  else if (is.ie4up) {
    var elt = eval('document.all.' + getElt.arguments
                       [getElt.arguments.length-1]);
    return(elt);
  }
}
