/* function AddSlide adds a slide.     */
/*                                     */
/* parameter name is the name of file, */
/* without the .htm extension.         */

function AddSlide(name)
{  last_slide++;
   if (name == null) name = last_slide - 
       slides_inserted + slides_deleted;
   else {   slides_inserted++;	}
   filename[last_slide] = name + ".htm";
}
